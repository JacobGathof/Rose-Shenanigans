#version 400 core

layout (location = 0) in vec2 vertices;
layout (location = 1) in vec2 texCoords;

out vec2 texCoords_;

uniform vec2 scale;
uniform vec2 pos;

uniform mat4 projectionMatrix;

void main(){
	
	gl_Position = projectionMatrix * vec4(vertices.x*scale.x + pos.x, vertices.y*scale.y + pos.y, 0.0 , 1.0);
	texCoords_ = texCoords;
	
}
