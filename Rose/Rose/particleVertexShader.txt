#version 400 core


layout (location = 0) in vec2 vertices;
layout (location = 1) in vec2 texCoords;

layout (location = 2) in vec2 position;
layout (location = 3) in vec3 color;


out vec3 color_;
out vec2 texCoords_;


void main(){

	gl_PointSize = 3.0;

	gl_Position = vec4(position+(vertices/16), 1.0, 1.0);

	color_ = color;
	texCoords_ = texCoords/2;

}
